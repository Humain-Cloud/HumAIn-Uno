    17→-- (Re-created with CREATE OR REPLACE so it works regardless of
    18→--  whether the 001 migration's version or supabase-setup.sql's
    19→--  version was applied. Both do the same thing.)
    20→-- ============================================================
    21→CREATE OR REPLACE FUNCTION public.update_updated_at_column()
    22→RETURNS TRIGGER AS $$
    23→BEGIN
    24→  NEW.updated_at = NOW();
    25→  RETURN NEW;
    26→END;

   827→    'grid',
   828→    'beginner'

   830→  ON CONFLICT (user_id) DO NOTHING;

   832→  -- Create default user settings

   834→  VALUES (

   837→    TRUE,
   838→    'none',
   839→    100
   840→  )
   841→  ON CONFLICT (user_id) DO NOTHING;
