   225→// ─── Explore Tab ────────────────────────────────────────────────────

   227→function ExploreTab({ userId }: { userId?: string }) {
   228→  const [models, setModels] = useState<LLMModel[]>([])
   229→  const [stats, setStats] = useState<ModelStats | null>(null)
   230→  const [loading, setLoading] = useState(true)
   231→  const [page, setPage] = useState(1)
   232→  const [totalPages, setTotalPages] = useState(1)
   233→  const [total, setTotal] = useState(0)
   234→  const pageSize = 24
   235→
   236→  // Filters
   237→  const [search, setSearch] = useState('')
   238→  const debouncedSearch = useDebounce(search, 400)
   239→  const [selectedOrg, setSelectedOrg] = useState<string>('')
   240→  const [selectedArena, setSelectedArena] = useState<string>('')
   241→  const [selectedLicense, setSelectedLicense] = useState<string>('')
   242→  const [selectedSort, setSelectedSort] = useState('rating')
   243→  const [showFilters, setShowFilters] = useState(false)
   244→  const [viewMode, setViewMode] = useState<'grid' | 'list'>('grid')
   245→  const [syncing, setSyncing] = useState(false)
   246→
   247→  const fetchModels = useCallback(async () => {
   248→    setLoading(true)
   249→    try {
   250→      const params = new URLSearchParams({
   251→        page: page.toString(),
   252→        pageSize: pageSize.toString(),
   253→        sort: selectedSort,
   254→      })
   255→      if (debouncedSearch) params.set('search', debouncedSearch)
   256→      if (selectedOrg) params.set('organization', selectedOrg)
   257→      if (selectedArena) params.set('arena', selectedArena)
   258→      if (selectedLicense) params.set('license', selectedLicense)
   259→
   260→      const res = await fetch(`/api/llm-models?${params}`)
   261→      const data = await res.json()
   262→      setModels(data.models || [])
   263→      setTotalPages(data.pagination?.totalPages || 1)
   264→      setTotal(data.pagination?.total || 0)
   265→    } catch (error) {
   266→      console.error('Failed to fetch models:', error)
   267→    } finally {
   268→      setLoading(false)
   269→    }
   270→  }, [page, debouncedSearch, selectedOrg, selectedArena, selectedLicense, selectedSort])
   271→
   272→  const fetchStats = useCallback(async () => {
   273→    try {
   274→      const res = await fetch('/api/llm-models/stats')
   275→      const data = await res.json()
   276→      setStats(data)
   277→    } catch (error) {
   278→      console.error('Failed to fetch stats:', error)
   279→    }
   280→  }, [])
   281→
   282→  useEffect(() => { fetchModels() }, [fetchModels])
   283→  useEffect(() => { fetchStats() }, [fetchStats])
   284→
   285→  const handleSync = async () => {
   286→    setSyncing(true)
   287→    try {
   288→      const res = await fetch('/api/llm-models/sync', { method: 'POST' })
   289→      const data = await res.json()
   290→      if (data.status === 'success') {
   291→        fetchModels()
   292→        fetchStats()
   293→      }
   294→    } catch (error) {
   295→      console.error('Sync failed:', error)
   296→    } finally {
   297→      setSyncing(false)
   298→    }
   299→  }
   300→
   301→  const activeFilterCount = [selectedOrg, selectedArena, selectedLicense, debouncedSearch].filter(Boolean).length
   302→
   303→  return (
   304→    <div className="space-y-6">
   305→      {/* Stats Banner */}
   306→      {stats && (
   307→        <div className="grid grid-cols-2 sm:grid-cols-4 gap-4">
   308→          <StatsCard icon={Cpu} label="Total Models" value={stats.totalModels.toString()} />
   309→          <StatsCard icon={Globe} label="Organizations" value={stats.totalOrganizations.toString()} />
   310→          <StatsCard icon={BarChart3} label="Avg Output $/M" value={`$${stats.pricing.output.avg?.toFixed(2) || 'N/A'}`} />
   311→          <StatsCard icon={TrendingUp} label="#1 Ranked" value={stats.topRated[0]?.name || 'N/A'} subtext={stats.topRated[0]?.organization} />
   312→        </div>
   313→      )}
   314→
   315→      {/* Search & Filter Bar */}
   316→      <div className="flex flex-col sm:flex-row gap-3">
   317→        <div className="relative flex-1">
   318→          <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
   319→          <Input
   320→            placeholder="Search models, organizations..."
   321→            value={search}
   322→            onChange={(e) => { setSearch(e.target.value); setPage(1) }}
   323→            className="pl-10 pr-4 h-10 rounded-xl border-border/60 bg-background"
   324→          />
   325→          {search && (
   326→            <button onClick={() => setSearch('')} className="absolute right-3 top-1/2 -translate-y-1/2">
   327→              <X className="h-4 w-4 text-muted-foreground hover:text-foreground" />
   328→            </button>
   329→          )}
   330→        </div>
   331→
   332→        <div className="flex gap-2">
   333→          <Button
   334→            variant="outline"
   335→            size="sm"
   336→            className="rounded-xl"
   337→            onClick={() => setShowFilters(!showFilters)}
   338→          >
   339→            <Filter className="h-4 w-4 mr-1" />
   340→            Filters
   341→            {activeFilterCount > 0 && (
   342→              <Badge variant="secondary" className="ml-1 h-5 w-5 p-0 flex items-center justify-center text-[10px] rounded-full bg-emerald-500 text-white">
   343→                {activeFilterCount}
   344→              </Badge>
   345→            )}
   346→          </Button>
   347→
   348→          <Select value={selectedSort} onValueChange={(v) => { setSelectedSort(v); setPage(1) }}>
   349→            <SelectTrigger className="w-[180px] h-10 rounded-xl">
   350→              <ArrowUpDown className="h-4 w-4 mr-1" />
   351→              <SelectValue placeholder="Sort by" />
   352→            </SelectTrigger>
   353→            <SelectContent>
   354→              {SORT_OPTIONS.map(o => (
   355→                <SelectItem key={o.value} value={o.value}>{o.label}</SelectItem>
   356→              ))}
   357→            </SelectContent>
   358→          </Select>
   359→
   360→          <div className="hidden sm:flex border rounded-xl overflow-hidden">
   361→            <Button variant={viewMode === 'grid' ? 'secondary' : 'ghost'} size="sm" className="rounded-none" onClick={() => setViewMode('grid')}>
   362→              <LayoutGrid className="h-4 w-4" />
   363→            </Button>
   364→            <Button variant={viewMode === 'list' ? 'secondary' : 'ghost'} size="sm" className="rounded-none" onClick={() => setViewMode('list')}>
   365→              <List className="h-4 w-4" />
   366→            </Button>
   367→          </div>
   368→
   369→          <Button variant="outline" size="sm" className="rounded-xl" onClick={handleSync} disabled={syncing}>
   370→            <RefreshCw className={`h-4 w-4 mr-1 ${syncing ? 'animate-spin' : ''}`} />
   371→            Sync
   372→          </Button>
   373→        </div>
   374→      </div>
   375→
   376→      {/* Filter Panel */}
   377→      {showFilters && (
   378→        <Card className="border-border/60">
   379→          <CardContent className="p-4">
   380→            <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
   381→              {/* Organization Filter */}
   382→              <div className="space-y-2">
   383→                <Label className="text-xs font-medium text-muted-foreground">Organization</Label>
   384→                <Select value={selectedOrg} onValueChange={(v) => { setSelectedOrg(v === '__all__' ? '' : v); setPage(1) }}>
   385→                  <SelectTrigger className="h-9 rounded-lg text-sm">
   386→                    <SelectValue placeholder="All organizations" />
   387→                  </SelectTrigger>
   388→                  <SelectContent className="max-h-60">
   389→                    <SelectItem value="__all__">All organizations</SelectItem>
   390→                    {stats?.organizations.map(o => (
   391→                      <SelectItem key={o.name} value={o.name}>
   392→                        {o.name} ({o.modelCount})
   393→                      </SelectItem>
   394→                    ))}
   395→                  </SelectContent>
   396→                </Select>
   397→              </div>
   398→
   399→              {/* Arena Category Filter */}
   400→              <div className="space-y-2">
   401→                <Label className="text-xs font-medium text-muted-foreground">Arena Category</Label>
   402→                <Select value={selectedArena} onValueChange={(v) => { setSelectedArena(v === '__all__' ? '' : v); setPage(1) }}>
   403→                  <SelectTrigger className="h-9 rounded-lg text-sm">
   404→                    <SelectValue placeholder="All arenas" />
   405→                  </SelectTrigger>
   406→                  <SelectContent>
   407→                    <SelectItem value="__all__">All arenas</SelectItem>
   408→                    {stats?.arenas.map(a => {
   409→                      const meta = ARENA_META[a.name]
   410→                      return (
   411→                        <SelectItem key={a.name} value={a.name}>
   412→                          {meta?.label || a.name} ({a.count})
   413→                        </SelectItem>
   414→                      )
   415→                    })}
   416→                  </SelectContent>
   417→                </Select>
   418→              </div>
   419→
   420→              {/* License Filter */}
   421→              <div className="space-y-2">
   422→                <Label className="text-xs font-medium text-muted-foreground">License</Label>
   423→                <Select value={selectedLicense} onValueChange={(v) => { setSelectedLicense(v === '__all__' ? '' : v); setPage(1) }}>
   424→                  <SelectTrigger className="h-9 rounded-lg text-sm">
   425→                    <SelectValue placeholder="All licenses" />
   426→                  </SelectTrigger>
   427→                  <SelectContent>
   428→                    <SelectItem value="__all__">All licenses</SelectItem>
   429→                    {stats?.licenses.map(l => (
   430→                      <SelectItem key={l.name} value={l.name}>
   431→                        {l.name} ({l.count})
   432→                      </SelectItem>
   433→                    ))}
   434→                  </SelectContent>
   435→                </Select>
   436→              </div>
   437→            </div>
   438→
   439→            {/* Active filters as chips */}
   440→            {activeFilterCount > 0 && (
   441→              <div className="flex items-center gap-2 mt-3 pt-3 border-t border-border/40">
   442→                <span className="text-xs text-muted-foreground">Active:</span>
   443→                {selectedOrg && <FilterChip label={selectedOrg} onRemove={() => setSelectedOrg('')} />}
   444→                {selectedArena && <FilterChip label={ARENA_META[selectedArena]?.label || selectedArena} onRemove={() => setSelectedArena('')} />}
   445→                {selectedLicense && <FilterChip label={selectedLicense} onRemove={() => setSelectedLicense('')} />}
   446→                <Button variant="ghost" size="sm" className="text-xs h-6" onClick={() => { setSelectedOrg(''); setSelectedArena(''); setSelectedLicense(''); setSearch(''); }}>
   447→                  Clear all
   448→                </Button>
   449→              </div>
   450→            )}
   451→          </CardContent>
   452→        </Card>
   453→      )}
   454→
   455→      {/* Results count */}
   456→      <div className="flex items-center justify-between">
   457→        <p className="text-sm text-muted-foreground">
   458→          {loading ? 'Loading...' : `${total} model${total !== 1 ? 's' : ''} found`}
   459→        </p>
   460→        {totalPages > 1 && (
   461→          <div className="flex items-center gap-2">
   462→            <Button variant="outline" size="sm" disabled={page <= 1} onClick={() => setPage(p => p - 1)}>Prev</Button>
   463→            <span className="text-sm text-muted-foreground">{page}/{totalPages}</span>
   464→            <Button variant="outline" size="sm" disabled={page >= totalPages} onClick={() => setPage(p => p + 1)}>Next</Button>
   465→          </div>
   466→        )}
   467→      </div>
   468→
   469→      {/* Models Grid/List */}
   470→      {loading ? (
   471→        <div className={viewMode === 'grid' ? 'grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4' : 'space-y-3'}>
   472→          {Array.from({ length: 6 }).map((_, i) => (
   473→            <Skeleton key={i} className="h-48 rounded-xl" />
   474→          ))}
   475→        </div>
   476→      ) : models.length === 0 ? (
   477→        <div className="text-center py-16">
   478→          <Cpu className="h-12 w-12 mx-auto text-muted-foreground/50 mb-4" />
   479→          <h3 className="text-lg font-medium text-muted-foreground">No models found</h3>
   480→          <p className="text-sm text-muted-foreground/70 mt-1">Try adjusting your search or filters</p>
   481→        </div>
   482→      ) : viewMode === 'grid' ? (
   483→        <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
   484→          {models.map(m => <ModelCard key={m.id} model={m} userId={userId} onBookmarkChange={fetchModels} />)}
   485→        </div>
   486→      ) : (
   487→        <div className="space-y-2">
   488→          {models.map(m => <ModelListItem key={m.id} model={m} userId={userId} onBookmarkChange={fetchModels} />)}
   489→        </div>
   490→      )}
   491→    </div>
   492→  )
   493→}
