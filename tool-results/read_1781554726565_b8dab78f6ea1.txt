     1→'use client'
     2→
     3→import { useState, useEffect, useCallback } from 'react'
     4→import { api } from '@/lib/api-client'
     5→import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
     6→import { Button } from '@/components/ui/button'
     7→import { Badge } from '@/components/ui/badge'
     8→import { Input } from '@/components/ui/input'
     9→import { Label } from '@/components/ui/label'
    10→import { Textarea } from '@/components/ui/textarea'
    11→import { Skeleton } from '@/components/ui/skeleton'
    12→import { Progress } from '@/components/ui/progress'
    13→import { Switch } from '@/components/ui/switch'
    14→import {
    15→  Select,
    16→  SelectContent,
    17→  SelectItem,
    18→  SelectTrigger,
    19→  SelectValue,
    20→} from '@/components/ui/select'
    21→import {
    22→  ArrowLeft,
    23→  ArrowRight,
    24→  Check,
    25→  Loader2,
    26→  AlertCircle,
    27→  X,
    28→  Paintbrush,
    29→  Settings2,
    30→  MessageSquare,
    31→  Eye,
    32→  Cpu,
    33→} from 'lucide-react'
    34→import { motion, AnimatePresence } from 'framer-motion'
    35→import { useToast } from '@/hooks/use-toast'
    36→
    37→interface ScratchWizardProps {
    38→  onCreated: () => void
    39→}
    40→
    41→const STEPS = [
    42→  { title: 'Basic Info', description: 'Name, description, and categorization', icon: Paintbrush },
    43→  { title: 'Framework & Model', description: 'Choose your tech stack', icon: Cpu },
    44→  { title: 'Agent Configuration', description: 'System prompt, tools, and behavior', icon: Settings2 },
    45→  { title: 'Review & Create', description: 'Review all settings and create', icon: Eye },
    46→]
    47→
    48→interface FormData {
    49→  name: string
    50→  description: string
    51→  category: string
    52→  industry: string
    53→  difficulty: string
    54→  language: string
    55→  framework: string
    56→  llm: string
    57→  systemPrompt: string
    58→  tools: string[]
    59→  tags: string[]
    60→  privacy: string
    61→  code: string
    62→}
    63→
    64→const initialFormData: FormData = {
    65→  name: '',
    66→  description: '',
    67→  category: '',
    68→  industry: '',
    69→  difficulty: 'intermediate',
    70→  language: 'python',
    71→  framework: '',
    72→  llm: '',
    73→  systemPrompt: '',
    74→  tools: [],
    75→  tags: [],
    76→  privacy: 'PUBLIC',
    77→  code: '',
    78→}
    79→
    80→export function ScratchWizard({ onCreated }: ScratchWizardProps) {
    81→  const { toast } = useToast()
    82→  const [step, setStep] = useState(0)
    83→  const [formData, setFormData] = useState<FormData>(initialFormData)
    84→  const [categories, setCategories] = useState<any[]>([])
    85→  const [llmModels, setLlmModels] = useState<any[]>([])
    86→  const [categoriesLoading, setCategoriesLoading] = useState(true)
    87→  const [modelsLoading, setModelsLoading] = useState(false)
    88→  const [publishing, setPublishing] = useState(false)
    89→  const [tagInput, setTagInput] = useState('')
    90→  const [toolInput, setToolInput] = useState('')
    91→  const [errors, setErrors] = useState<Record<string, string>>({})
    92→
    93→  // Load categories
    94→  useEffect(() => {
    95→    api.categories.list().then((data: any) => {
    96→      setCategories(Array.isArray(data) ? data : [])
    97→      setCategoriesLoading(false)
    98→    }).catch(() => setCategoriesLoading(false))
    99→  }, [])
   100→
   101→  // Load LLM models when step 1 is reached
   102→  useEffect(() => {
   103→    if (step >= 1 && llmModels.length === 0) {
   104→      setModelsLoading(true)
   105→      fetch('/api/llm-models?pageSize=50&sort=rating&userSelectable=true')
   106→        .then(res => res.json())
   107→        .then(data => {
   108→          const models = data?.models || []
   109→          setLlmModels(models)
   110→        })
   111→        .catch(console.error)
   112→        .finally(() => setModelsLoading(false))
   113→    }
   114→  }, [step, llmModels.length])
   115→
   116→  const updateField = useCallback((field: keyof FormData, value: any) => {
   117→    setFormData(prev => ({ ...prev, [field]: value }))
   118→    // Clear error for that field
   119→    setErrors(prev => {
   120→      const next = { ...prev }
   121→      delete next[field]
   122→      return next
   123→    })
   124→  }, [])
   125→
   126→  const validateStep = (s: number): boolean => {
   127→    const newErrors: Record<string, string> = {}
   128→    if (s === 0) {
   129→      if (!formData.name.trim() || formData.name.trim().length < 3) {
   130→        newErrors.name = 'Agent name must be at least 3 characters'
   131→      }
   132→      if (!formData.description.trim() || formData.description.trim().length < 10) {
   133→        newErrors.description = 'Description must be at least 10 characters'
   134→      }
   135→      if (!formData.category) {
   136→        newErrors.category = 'Please select a category'
   137→      }
   138→    }
   139→    if (s === 1) {
   140→      if (!formData.framework) {
   141→        newErrors.framework = 'Please select a framework'
   142→      }
   143→    }
   144→    setErrors(newErrors)
   145→    return Object.keys(newErrors).length === 0
   146→  }
   147→
   148→  const handleNext = () => {
   149→    if (validateStep(step)) {
   150→      setStep(prev => Math.min(prev + 1, STEPS.length - 1))
   151→    }
   152→  }
   153→
   154→  const handlePrev = () => {
   155→    setStep(prev => Math.max(prev - 1, 0))
   156→  }
   157→
   158→  const handleAddTag = () => {
   159→    if (!tagInput.trim()) return
   160→    if (!formData.tags.includes(tagInput.trim())) {
   161→      updateField('tags', [...formData.tags, tagInput.trim()])
   162→    }
   163→    setTagInput('')
   164→  }
   165→
   166→  const handleRemoveTag = (tag: string) => {
   167→    updateField('tags', formData.tags.filter(t => t !== tag))
   168→  }
   169→
   170→  const handleAddTool = () => {
   171→    if (!toolInput.trim()) return
   172→    if (!formData.tools.includes(toolInput.trim())) {
   173→      updateField('tools', [...formData.tools, toolInput.trim()])
   174→    }
   175→    setToolInput('')
   176→  }
   177→
   178→  const handleRemoveTool = (tool: string) => {
   179→    updateField('tools', formData.tools.filter(t => t !== tool))
   180→  }
   181→
   182→  const handlePublish = async () => {
   183→    setPublishing(true)
   184→    try {
   185→      const cat = categories.find(c => c.id === formData.category || c.name === formData.category)
   186→      const readme = `# ${formData.name}\n\n${formData.description}\n\n## Configuration\n\n- **Framework**: ${formData.framework}\n- **LLM**: ${formData.llm || 'Not specified'}\n- **Language**: ${formData.language}\n- **Difficulty**: ${formData.difficulty}\n- **Industry**: ${formData.industry || 'General'}\n\n## System Prompt\n\n${formData.systemPrompt || 'No system prompt defined.'}\n\n## Tools\n\n${formData.tools.length > 0 ? formData.tools.map(t => `- ${t}`).join('\n') : 'No tools configured.'}\n\n## Tags\n\n${formData.tags.length > 0 ? formData.tags.map(t => `\`${t}\``).join(', ') : 'No tags'}\n`
   187→
   188→      const result: any = await api.agents.create({
   189→        name: formData.name,
   190→        description: formData.description,
   191→        categoryId: cat?.id || formData.category || categories[0]?.id,
   192→        privacy: formData.privacy,
   193→        source: 'scratch',
   194→        readme,
   195→        code: formData.code || null,
   196→        tags: formData.tags,
   197→        framework: formData.framework,
   198→        llm: formData.llm,
   199→        industry: formData.industry,
   200→        difficulty: formData.difficulty,
   201→        language: formData.language,
   202→      })
   203→
   204→      toast({ title: 'Agent created!', description: `${formData.name} has been created successfully.` })
   205→      onCreated()
   206→    } catch (err: any) {
   207→      toast({ title: 'Creation failed', description: err.message || 'Please try again', variant: 'destructive' })
   208→    } finally {
   209→      setPublishing(false)
   210→    }
   211→  }
   212→
   213→  const progress = ((step + 1) / STEPS.length) * 100
   214→
   215→  // Generate scaffolded code based on framework
   216→  const generateCode = (): string => {
   217→    const fw = formData.framework
   218→    const name = formData.name.replace(/[^a-zA-Z0-9]/g, '') || 'MyAgent'
   219→    const llm = formData.llm || 'gpt-4o'
   220→
   221→    switch (fw) {
   222→      case 'langgraph':
   223→        return `from langgraph.graph import StateGraph, END
   224→from typing import TypedDict, Annotated
   225→import operator
   226→
   227→class ${name}State(TypedDict):
   228→    input: str
   229→    output: str
   230→    messages: Annotated[list, operator.add]
   231→
   232→def process_input(state: ${name}State) -> dict:
   233→    """Process the input and generate a response."""
   234→    return {"output": "Response generated", "messages": ["Processing complete"]}
   235→
   236→workflow = StateGraph(${name}State)
   237→workflow.add_node("process", process_input)
   238→workflow.add_edge("__start__", "process")
   239→workflow.add_edge("process", END)
   240→
   241→app = workflow.compile()`
   242→
   243→      case 'crewai':
   244→        return `from crewai import Agent, Task, Crew, Process
   245→
   246→${name}_agent = Agent(
   247→    role="${formData.name}",
   248→    goal="${formData.description}",
   249→    backstory="An expert AI agent designed for specific tasks",
   250→    verbose=True,
   251→    allow_delegation=False,
   252→    llm="${llm}",
   253→)
   254→
   255→main_task = Task(
   256→    description="Execute the main workflow",
   257→    agent=${name}_agent,
   258→    expected_output="Completed task result",
   259→)
   260→
   261→crew = Crew(
   262→    agents=[${name}_agent],
   263→    tasks=[main_task],
   264→    process=Process.sequential,
   265→)
   266→
   267→result = crew.kickoff()`
   268→
   269→      case 'autogen':
   270→        return `import autogen
   271→
   272→config_list = autogen.config_list_from_json("OAI_CONFIG_LIST")
   273→llm_config = {"config_list": config_list, "timeout": 120}
   274→
   275→${name}_agent = autogen.AssistantAgent(
   276→    name="${name}",
   277→    llm_config=llm_config,
   278→    system_message="""${formData.systemPrompt || `You are ${formData.name}. ${formData.description}`}""",
   279→)
   280→
   281→user_proxy = autogen.UserProxyAgent(
   282→    name="User",
   283→    human_input_mode="TERMINATE",
   284→    max_consecutive_auto_reply=5,
   285→)
   286→
   287→user_proxy.initiate_chat(${name}_agent, message="Hello!")`
   288→
   289→      case 'agno':
   290→        return `from agno.agent import Agent
   291→from agno.models.openai import OpenAIChat
   292→
   293→${name}_agent = Agent(
   294→    name="${name}",
   295→    model=OpenAIChat(id="${llm}"),
   296→    instructions=[
   297→        "${formData.systemPrompt || formData.description}",
   298→        "Always provide thorough and helpful responses.",
   299→    ],
   300→    show_tool_calls=True,
   301→    markdown=True,
   302→)
   303→
   304→${name}_agent.print_response("Hello! How can you help me today?")`
   305→
   306→      case 'llamaindex':
   307→        return `from llama_index.core import VectorStoreIndex, SimpleDirectoryReader, Settings
   308→from llama_index.llms.openai import OpenAI
   309→from llama_index.core.agent import ReActAgent
   310→
   311→Settings.llm = OpenAI(model="${llm}")
   312→
   313→documents = SimpleDirectoryReader("./data").load_data()
   314→index = VectorStoreIndex.from_documents(documents)
   315→query_engine = index.as_query_engine()
   316→
   317→agent = ReActAgent.from_tools(
   318→    [],
   319→    llm=OpenAI(model="${llm}"),
   320→    verbose=True,
   321→)
   322→
   323→response = agent.chat("Hello!")`
   324→
   325→      default:
   326→        return `# ${formData.name}\n# Framework: ${fw}\n# LLM: ${llm}\n\n# TODO: Implement your agent logic here`
   327→    }
   328→  }
   329→
   330→  return (
   331→    <div>
   332→      {/* Header */}
   333→      <div className="mb-6">
   334→        <div className="flex items-center justify-between">
   335→          <div>
   336→            <h2 className="text-2xl font-bold">From Scratch</h2>
   337→            <p className="text-sm text-muted-foreground mt-1">{STEPS[step].description}</p>
   338→          </div>
   339→          <Badge variant="outline" className="text-sm font-medium px-3 py-1">
   340→            {Math.round(progress)}% complete
   341→          </Badge>
   342→        </div>
   343→        <Progress value={progress} className="mt-4 h-2" />
   344→      </div>
   345→
   346→      {/* Step Indicators */}
   347→      <div className="flex items-center gap-2 mb-8 overflow-x-auto pb-2">
   348→        {STEPS.map((s, i) => (
   349→          <div key={i} className="flex items-center gap-2">
   350→            <button
   351→              onClick={() => i < step && setStep(i)}
   352→              className={`flex items-center gap-2 px-3 py-2 rounded-lg text-sm font-medium transition-all whitespace-nowrap ${
   353→                i === step
   354→                  ? 'bg-emerald-100 text-emerald-700 dark:bg-emerald-900/30 dark:text-emerald-300'
   355→                  : i < step
   356→                  ? 'bg-gray-100 text-gray-600 dark:bg-gray-800 dark:text-gray-400 cursor-pointer hover:bg-gray-200 dark:hover:bg-gray-700'
   357→                  : 'text-muted-foreground'
   358→              }`}
   359→            >
   360→              {i < step ? (
   361→                <Check className="h-4 w-4 text-emerald-600" />
   362→              ) : (
   363→                <s.icon className="h-4 w-4" />
   364→              )}
   365→              <span className="hidden sm:inline">{s.title}</span>
   366→              <span className="sm:hidden">{i + 1}</span>
   367→            </button>
   368→            {i < STEPS.length - 1 && (
   369→              <div className={`h-0.5 w-6 ${i < step ? 'bg-emerald-400' : 'bg-gray-200 dark:bg-gray-700'}`} />
   370→            )}
   371→          </div>
   372→        ))}
   373→      </div>
   374→
   375→      {/* Step Content */}
   376→      <AnimatePresence mode="wait">
   377→        <motion.div
   378→          key={step}
   379→          initial={{ opacity: 0, x: 20 }}
   380→          animate={{ opacity: 1, x: 0 }}
   381→          exit={{ opacity: 0, x: -20 }}
   382→          transition={{ duration: 0.25 }}
   383→        >
   384→          {/* Step 0: Basic Info */}
   385→          {step === 0 && (
   386→            <Card className="rounded-2xl">
   387→              <CardContent className="p-6 space-y-5">
   388→                <div className="space-y-2">
   389→                  <Label htmlFor="name" className="flex items-center gap-1">
   390→                    Agent Name *
   391→                    {errors.name && (
   392→                      <span className="text-rose-500 text-xs flex items-center gap-1">
   393→                        <AlertCircle className="h-3 w-3" /> {errors.name}
   394→                      </span>
   395→                    )}
   396→                  </Label>
   397→                  <Input
   398→                    id="name"
   399→                    placeholder="My Awesome Agent"
   400→                    value={formData.name}
   401→                    onChange={e => updateField('name', e.target.value)}
   402→                    className={errors.name ? 'border-rose-300' : ''}
   403→                  />
   404→                </div>
   405→
   406→                <div className="space-y-2">
   407→                  <Label htmlFor="description" className="flex items-center gap-1">
   408→                    Description *
   409→                    {errors.description && (
   410→                      <span className="text-rose-500 text-xs flex items-center gap-1">
   411→                        <AlertCircle className="h-3 w-3" /> {errors.description}
   412→                      </span>
   413→                    )}
   414→                  </Label>
   415→                  <Textarea
   416→                    id="description"
   417→                    placeholder="Describe what your agent does in detail..."
   418→                    value={formData.description}
   419→                    onChange={e => updateField('description', e.target.value)}
   420→                    className={`min-h-[100px] ${errors.description ? 'border-rose-300' : ''}`}
   421→                  />
   422→                </div>
   423→
   424→                <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   425→                  <div className="space-y-2">
   426→                    <Label className="flex items-center gap-1">
   427→                      Category *
   428→                      {errors.category && (
   429→                        <span className="text-rose-500 text-xs">{errors.category}</span>
   430→                      )}
   431→                    </Label>
   432→                    {categoriesLoading ? (
   433→                      <Skeleton className="h-10 w-full" />
   434→                    ) : (
   435→                      <Select
   436→                        value={formData.category}
   437→                        onValueChange={v => updateField('category', v)}
   438→                      >
   439→                        <SelectTrigger className={errors.category ? 'border-rose-300' : ''}>
   440→                          <SelectValue placeholder="Select category" />
   441→                        </SelectTrigger>
   442→                        <SelectContent>
   443→                          {categories.map((cat: any) => (
   444→                            <SelectItem key={cat.id} value={cat.id}>
   445→                              {cat.name}
   446→                            </SelectItem>
   447→                          ))}
   448→                        </SelectContent>
   449→                      </Select>
   450→                    )}
   451→                  </div>
   452→
   453→                  <div className="space-y-2">
   454→                    <Label>Industry</Label>
   455→                    <Select
   456→                      value={formData.industry}
   457→                      onValueChange={v => updateField('industry', v)}
   458→                    >
   459→                      <SelectTrigger>
   460→                        <SelectValue placeholder="Select industry" />
   461→                      </SelectTrigger>
   462→                      <SelectContent>
   463→                        <SelectItem value="technology">Technology</SelectItem>
   464→                        <SelectItem value="healthcare">Healthcare</SelectItem>
   465→                        <SelectItem value="finance">Finance</SelectItem>
   466→                        <SelectItem value="education">Education</SelectItem>
   467→                        <SelectItem value="marketing">Marketing</SelectItem>
   468→                        <SelectItem value="legal">Legal</SelectItem>
   469→                        <SelectItem value="ecommerce">E-Commerce</SelectItem>
   470→                        <SelectItem value="general">General</SelectItem>
   471→                      </SelectContent>
   472→                    </Select>
   473→                  </div>
   474→
   475→                  <div className="space-y-2">
   476→                    <Label>Difficulty</Label>
   477→                    <Select
   478→                      value={formData.difficulty}
   479→                      onValueChange={v => updateField('difficulty', v)}
   480→                    >
   481→                      <SelectTrigger>
   482→                        <SelectValue />
   483→                      </SelectTrigger>
   484→                      <SelectContent>
   485→                        <SelectItem value="beginner">Beginner</SelectItem>
   486→                        <SelectItem value="intermediate">Intermediate</SelectItem>
   487→                        <SelectItem value="advanced">Advanced</SelectItem>
   488→                      </SelectContent>
   489→                    </Select>
   490→                  </div>
   491→
   492→                  <div className="space-y-2">
   493→                    <Label>Language</Label>
   494→                    <Select
   495→                      value={formData.language}
   496→                      onValueChange={v => updateField('language', v)}
   497→                    >
   498→                      <SelectTrigger>
   499→                        <SelectValue />
   500→                      </SelectTrigger>
   501→                      <SelectContent>
   502→                        <SelectItem value="python">Python</SelectItem>
   503→                        <SelectItem value="typescript">TypeScript</SelectItem>
   504→                      </SelectContent>
   505→                    </Select>
   506→                  </div>
   507→                </div>
   508→              </CardContent>
   509→            </Card>
   510→          )}
   511→
   512→          {/* Step 1: Framework & Model */}
   513→          {step === 1 && (
   514→            <div className="space-y-6">
   515→              <Card className="rounded-2xl">
   516→                <CardHeader className="pb-3">
   517→                  <CardTitle className="text-lg">Framework</CardTitle>
   518→                </CardHeader>
   519→                <CardContent className="p-6 pt-0">
   520→                  {errors.framework && (
   521→                    <p className="text-rose-500 text-xs mb-3 flex items-center gap-1">
   522→                      <AlertCircle className="h-3 w-3" /> {errors.framework}
   523→                    </p>
   524→                  )}
   525→                  <div className="grid grid-cols-2 sm:grid-cols-5 gap-3">
   526→                    {[
   527→                      { id: 'langgraph', label: 'LangGraph', color: 'emerald' },
   528→                      { id: 'crewai', label: 'CrewAI', color: 'amber' },
   529→                      { id: 'autogen', label: 'AutoGen', color: 'rose' },
   530→                      { id: 'agno', label: 'Agno', color: 'violet' },
   531→                      { id: 'llamaindex', label: 'LlamaIndex', color: 'teal' },
   532→                    ].map(fw => (
   533→                      <button
   534→                        key={fw.id}
   535→                        onClick={() => updateField('framework', fw.id)}
   536→                        className={`relative p-4 rounded-xl border-2 transition-all text-center hover:shadow-md ${
   537→                          formData.framework === fw.id
   538→                            ? `border-${fw.color}-400 bg-${fw.color}-50 dark:bg-${fw.color}-900/20`
   539→                            : 'border-gray-200 dark:border-gray-700 hover:border-gray-300'
   540→                        }`}
   541→                      >
   542→                        {formData.framework === fw.id && (
   543→                          <Check className="absolute top-2 right-2 h-4 w-4 text-emerald-600" />
   544→                        )}
   545→                        <span className="text-sm font-semibold">{fw.label}</span>
   546→                      </button>
   547→                    ))}
   548→                  </div>
   549→                </CardContent>
   550→              </Card>
   551→
   552→              <Card className="rounded-2xl">
   553→                <CardHeader className="pb-3">
   554→                  <CardTitle className="text-lg">LLM Model</CardTitle>
   555→                </CardHeader>
   556→                <CardContent className="p-6 pt-0">
   557→                  {modelsLoading ? (
   558→                    <div className="space-y-3">
   559→                      {Array.from({ length: 3 }).map((_, i) => (
   560→                        <Skeleton key={i} className="h-12 w-full rounded-lg" />
   561→                      ))}
   562→                    </div>
   563→                  ) : (
   564→                    <>
   565→                      <Select
   566→                        value={formData.llm}
   567→                        onValueChange={v => updateField('llm', v)}
   568→                      >
   569→                        <SelectTrigger>
   570→                          <SelectValue placeholder="Select LLM model" />
   571→                        </SelectTrigger>
   572→                        <SelectContent className="max-h-64">
   573→                          {llmModels.length > 0 ? (
   574→                            llmModels.map((model: any) => (
   575→                              <SelectItem key={model.id} value={model.name}>
   576→                                <span className="flex items-center gap-2">
   577→                                  {model.name}
   578→                                  <span className="text-xs text-muted-foreground">({model.organization})</span>
   579→                                </span>
