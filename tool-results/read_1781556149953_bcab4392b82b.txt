   210→   210→function formatRelativeTime(dateStr: string): string {
   211→   211→  try {
   212→   212→    const date = new Date(dateStr)
   213→   213→    const now = new Date()
   214→   214→    const diffMs = now.getTime() - date.getTime()
   215→   215→    const diffMins = Math.floor(diffMs / 60000)
   216→   216→    const diffHours = Math.floor(diffMs / 3600000)
   217→   217→    const diffDays = Math.floor(diffMs / 86400000)
   218→   218→
   219→   219→    if (diffMins < 1) return 'Just now'
   220→   220→    if (diffMins < 60) return `${diffMins}m ago`
   221→   221→    if (diffHours < 24) return `${diffHours}h ago`
   222→   222→    if (diffDays < 7) return `${diffDays}d ago`
   223→   223→    return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' })
   224→   224→  } catch {
   225→   225→    return dateStr
   226→   226→  }
   227→   227→}

   231→   231→function MiniSparkline({ data, color = '#10b981', width = 60, height = 24 }: { data: number[]; color?: string; width?: number; height?: number }) {
   232→   232→  if (!data || data.length < 2) return null
   233→   233→  const max = Math.max(...data)
   234→   234→  const min = Math.min(...data)
   235→   235→  const range = max - min || 1
   236→   236→
   237→   237→  const points = data.map((v, i) => {
   238→   238→    const x = (i / (data.length - 1)) * width
   239→   239→    const y = height - ((v - min) / range) * (height - 4) - 2
   240→   240→    return `${x},${y}`
   241→   241→  }).join(' ')
   242→   242→
   243→   243→  return (
   244→   244→    <svg width={width} height={height} className="inline-block ml-1">
   245→   245→      <polyline
   246→   246→        points={points}
   247→   247→        fill="none"
   248→   248→        stroke={color}
   249→   249→        strokeWidth="1.5"
   250→   250→        strokeLinecap="round"
   251→   251→        strokeLinejoin="round"
   252→   252→      />
   253→   253→    </svg>
   254→   254→  )
   255→   255→}
