   272→export default function OnboardingPage() {
   273→  const router = useRouter()
   274→  const { user: authUser, loading: authLoading } = useAuth()
   275→  const [currentStep, setCurrentStep] = useState(1)
   276→  const [direction, setDirection] = useState(1)
   277→  const [saving, setSaving] = useState(false)
   278→  const [completed, setCompleted] = useState(false)
   279→  const [authChecked, setAuthChecked] = useState(false)
   280→  const [userEmail, setUserEmail] = useState<string | null>(null)
   281→
   282→  // Onboarding form data
   283→  const [formData, setFormData] = useState<OnboardingData>({
   284→    full_name: '',
   285→    company: '',
   286→    job_title: '',
   287→    preferred_frameworks: [],
   288→    preferred_industries: [],
   289→    experience_level: 'beginner',
   290→    use_cases: [],
   291→    interests: [],
   292→  })
   293→
   294→  // Additional profile fields
   295→  const [location, setLocation] = useState('')
   296→
   297→  // Check Supabase auth state
   298→  useEffect(() => {
   299→    const checkAuth = async () => {
   300→      try {
   301→        const supabase = getSupabaseBrowserClient()
   302→        const { data: { user } } = await supabase.auth.getUser()
   303→        if (user) {
   304→          setUserEmail(user.email ?? null)
   305→          setFormData((prev) => ({
   306→            ...prev,
   307→            full_name: prev.full_name || user.user_metadata?.full_name || user.user_metadata?.name || '',
   308→          }))
   309→        }
   310→        // If no Supabase user, check NextAuth session
   311→      } catch {
   312→        // Supabase not configured - continue with NextAuth session
   313→      }
   314→      setAuthChecked(true)
   315→    }
   316→    checkAuth()
   317→  }, [])
   318→
   319→  // Pre-fill from Supabase auth
   320→  useEffect(() => {
   321→    if (authUser) {
   322→      setFormData((prev) => ({
   323→        ...prev,
   324→        full_name: prev.full_name || authUser.user_metadata?.full_name || authUser.user_metadata?.name || '',
   325→      }))
   326→      if (!userEmail && authUser.email) {
   327→        setUserEmail(authUser.email)
   328→      }
   329→    }
   330→  }, [authUser, userEmail])
   331→
   332→  // Save progress to Supabase
   333→  const saveProgress = useCallback(
   334→    async (step: number, isComplete = false) => {
   335→      setSaving(true)
   336→      try {
   337→        const userId = authUser?.id || userEmail || 'demo-user'
   338→        const profileData: Record<string, unknown> = {}
   339→        const preferencesData: Record<string, unknown> = {}
   340→
   341→        if (step >= 1) {
   342→          profileData.full_name = formData.full_name
   343→          profileData.company = formData.company
   344→          profileData.job_title = formData.job_title
   345→          profileData.location = location
   346→          profileData.email = userEmail || authUser?.email || ''
   347→        }
   348→        if (step >= 2) {
   349→          profileData.preferred_framework = formData.experience_level
   350→          preferencesData.experience_level = formData.experience_level
   351→        }
   352→        if (step >= 3) {
   353→          preferencesData.framework_filter = formData.preferred_frameworks
   354→        }
   355→        if (step >= 4) {
   356→          preferencesData.industry_filter = formData.preferred_industries
   357→          profileData.preferred_industry = formData.preferred_industries[0] || null
   358→        }
   359→        if (step >= 5) {
   360→          preferencesData.use_cases = formData.use_cases
   361→          preferencesData.interests = formData.interests
   362→        }
   363→
   364→        await fetch('/api/onboarding', {
   365→          method: 'POST',
   366→          headers: { 'Content-Type': 'application/json' },
   367→          body: JSON.stringify({
   368→            userId,
   369→            step,
   370→            profileData,
   371→            preferencesData,
   372→            onboardingCompleted: isComplete,
   373→          }),
   374→        })
   375→      } catch (error) {
   376→        console.error('Failed to save progress:', error)
   377→      } finally {
   378→        setSaving(false)
   379→      }
   380→    },
   381→    [formData, location, authUser, userEmail]
   382→  )
   383→
   384→  // Navigation handlers
   385→  const goNext = useCallback(async () => {
   386→    if (currentStep < TOTAL_STEPS) {
   387→      await saveProgress(currentStep)
   388→      setDirection(1)
   389→      setCurrentStep((s) => s + 1)
   390→    } else {
   391→      // Complete onboarding
   392→      await saveProgress(currentStep, true)
   393→      setCompleted(true)
   394→      setTimeout(() => {
   395→        router.push('/')
   396→      }, 2500)
   397→    }
   398→  }, [currentStep, saveProgress, router])
   399→
   400→  const goBack = useCallback(() => {
   401→    if (currentStep > 1) {
   402→      setDirection(-1)
   403→      setCurrentStep((s) => s - 1)
   404→    }
   405→  }, [currentStep])
   406→
   407→  const goSkip = useCallback(async () => {
   408→    await saveProgress(currentStep)
   409→    if (currentStep < TOTAL_STEPS) {
   410→      setDirection(1)
   411→      setCurrentStep((s) => s + 1)
   412→    } else {
   413→      await saveProgress(currentStep, true)
   414→      setCompleted(true)
   415→      setTimeout(() => {
   416→        router.push('/')
   417→      }, 2500)
   418→    }
   419→  }, [currentStep, saveProgress, router])
   420→
   421→  const skipAll = useCallback(async () => {
   422→    await saveProgress(currentStep, true)
   423→    setCompleted(true)
   424→    setTimeout(() => {
   425→      router.push('/')
   426→    }, 1500)
   427→  }, [currentStep, saveProgress, router])
   428→
   429→  // Validation
   430→  const isNextDisabled = (() => {
   431→    switch (currentStep) {
   432→      case 1:
   433→        return !formData.full_name.trim()
   434→      case 2:
   435→        return false // Experience level always has a default
   436→      case 3:
   437→        return formData.preferred_frameworks.length === 0
   438→      case 4:
   439→        return false // Industries are optional
   440→      case 5:
   441→        return false // Use cases are optional
   442→      default:
   443→        return false
   444→    }
   445→  })()
   446→
   447→  // ─── Step Renderers ──────────────────────────────────────────────────────
   448→
   449→  const renderStep1 = () => (
   450→    <div className="space-y-6">
   451→      {/* Welcome banner */}
   452→      <div className="rounded-xl bg-gradient-to-br from-emerald-50 to-emerald-100/50 dark:from-emerald-950/30 dark:to-emerald-900/20 p-5 sm:p-6 border border-emerald-200/50 dark:border-emerald-800/30">
   453→        <div className="flex items-start gap-4">
   454→          <div className="flex-shrink-0 h-12 w-12 rounded-xl bg-emerald-500 text-white flex items-center justify-center">
   455→            <Sparkles className="h-6 w-6" />
   456→          </div>
   457→          <div>
   458→            <h2 className="text-lg font-semibold text-emerald-900 dark:text-emerald-100">
   459→              Welcome to Humain-Uno! 🎉
   460→            </h2>
   461→            <p className="text-sm text-emerald-700 dark:text-emerald-300 mt-1">
   462→              {userEmail || authUser?.email
   463→                ? `Signed in as ${userEmail || authUser?.email}`
   464→                : 'Let\'s set up your personalized experience'}
   465→            </p>
   466→          </div>
   467→        </div>
   468→      </div>
   469→
   470→      {/* Profile form */}
   471→      <div className="grid gap-5">
   472→        <div className="space-y-2">
   473→          <Label htmlFor="full-name" className="text-sm font-medium">
   474→            Full Name <span className="text-emerald-500">*</span>
   475→          </Label>
   476→          <div className="relative">
   477→            <UserPlus className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
   478→            <Input
   479→              id="full-name"
   480→              placeholder="Enter your full name"
   481→              value={formData.full_name}
   482→              onChange={(e) =>
   483→                setFormData((d) => ({ ...d, full_name: e.target.value }))
   484→              }
   485→              className="pl-10 input-focus"
   486→            />
   487→          </div>
   488→        </div>
   489→
   490→        <div className="grid grid-cols-1 sm:grid-cols-2 gap-5">
   491→          <div className="space-y-2">
   492→            <Label htmlFor="company" className="text-sm font-medium">
   493→              <span className="flex items-center gap-1.5">

   495→                Company / Organization
   496→              </span>
   497→            </Label>
   498→            <Input
   499→              id="company"
   500→              placeholder="Acme Inc."
   501→              value={formData.company}
   502→              onChange={(e) =>
   503→                setFormData((d) => ({ ...d, company: e.target.value }))

   505→              className="input-focus"
   506→            />
   507→          </div>

   523→            />

   526→
   527→        <div className="space-y-2">
   528→          <Label htmlFor="location" className="text-sm font-medium">
